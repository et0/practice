## \# 002 
```go
func f1(s []int) {       // [1, 10, 3, 40]
    s[1] = 20            // [1, 20, 3, 40]
    s = append(s, 80)    // [1, 20, 3, 40, 80]
}

func f2(s *[]int) {
    (*s)[1] = 10         // [1, 10, 3]      cap=3
    *s = append(*s, 40)  // [1, 10, 3, 40]  cap=6
}

s := []int{1, 2, 3}
f2(&s)                   // [1, 10, 3, 40]
f1(s)                    // [1, 10, 3, 40]
fmt.Println(s)           // [1, 20, 3, 40]
```
